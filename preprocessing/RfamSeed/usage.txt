# First you need to download Rfam.seed sequence from Rfam website.


# The following command creates a stockhold format file of RF00001 family.
python _makeRfamSeedSto.py RF00001 Rfam.seed > rfam_out\rfam_stk\RF00001.stk
# Create .stk files for additional families (examples):
python _makeRfamSeedSto.py RF03064 Rfam.seed > rfam_out\rfam_stk\RF03064.stk
python _makeRfamSeedSto.py RF02914 Rfam.seed > rfam_out\rfam_stk\RF02914.stk
python _makeRfamSeedSto.py RF03156 Rfam.seed > rfam_out\rfam_stk\RF03156.stk


# The following command creates unaligned and aligned training data.
python _make_input_consG4.py rfam_out\rfam_stk\RF03156.stk rfam_out\rfam_unaligned\RF03156_unaligned.txt rfam_out\rfam_aligned\RF03156_aligned.txt
#多文件示例
python _make_input_consG4.py rfam_out\rfam_stk\RF03064.stk rfam_out\rfam_unaligned\RF03064_unaligned.txt rfam_out\rfam_aligned\RF03064_aligned.txt
python _make_input_consG4.py rfam_out\rfam_stk\RF02914.stk rfam_out\rfam_unaligned\RF02914_unaligned.txt rfam_out\rfam_aligned\RF02914_aligned.txt
python _make_input_consG4.py rfam_out\rfam_stk\RF03156.stk rfam_out\rfam_unaligned\RF03156_unaligned.txt rfam_out\rfam_aligned\RF03156_aligned.txt


# 生成矩阵
python _make_input_matrix.py rfam_out\rfam_unaligned\RF00001_unaligned.txt ..\..\output\matrices_unaligned --aggregate
python _make_input_matrix.py rfam_out\rfam_aligned\RF00001_aligned.txt ..\..\output\matrices_aligned --aggregate
# PowerShell 多文件示例（反引号续行）：
python _make_input_matrix.py `
rfam_out\rfam_unaligned\RF02914_unaligned.txt `
rfam_out\rfam_unaligned\RF03064_unaligned.txt `
rfam_out\rfam_unaligned\RF03156_unaligned.txt `
 ..\..\output\matrices_unaligned --aggregate
